-- ============================================================
-- SCHEMA CREATION
-- ============================================================

-- 1. Films master table
CREATE TABLE IF NOT EXISTS films (
    film_id         SERIAL PRIMARY KEY,
    title           VARCHAR(200) NOT NULL,
    year            INTEGER,
    genre           VARCHAR(100),
    subgenre        VARCHAR(100),
    director        VARCHAR(150),
    production_budget_usd BIGINT,
    mpaa_rating     VARCHAR(10),
    runtime_minutes INTEGER,
    language        VARCHAR(50) DEFAULT 'English',
    studio          VARCHAR(150),
    distributor     VARCHAR(150),
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 2. Global box office performance by genre and territory
CREATE TABLE IF NOT EXISTS box_office_performance (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    territory       VARCHAR(100),
    territory_code  CHAR(2),
    opening_weekend_usd BIGINT,
    total_gross_usd BIGINT,
    weeks_in_release INTEGER,
    screens_count   INTEGER,
    avg_ticket_price_usd NUMERIC(6,2),
    market_share_pct NUMERIC(5,2),
    release_date    DATE,
    distributor_local VARCHAR(150),
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 3. Actor Q-scores and social media reach
CREATE TABLE IF NOT EXISTS actor_metrics (
    actor_id        SERIAL PRIMARY KEY,
    name            VARCHAR(150) NOT NULL,
    nationality     VARCHAR(100),
    q_score_us      NUMERIC(5,2),       -- 0-100 familiarity/appeal composite
    q_score_global  NUMERIC(5,2),
    instagram_followers_m NUMERIC(6,2), -- millions
    twitter_followers_m   NUMERIC(6,2),
    tiktok_followers_m    NUMERIC(6,2),
    youtube_subscribers_m NUMERIC(6,2),
    google_search_index   INTEGER,      -- relative search interest 0-100
    award_wins      INTEGER DEFAULT 0,
    award_nominations INTEGER DEFAULT 0,
    franchise_value_usd BIGINT,         -- estimated box office pull
    last_updated    DATE
);

-- 4. Film-Actor mapping
CREATE TABLE IF NOT EXISTS film_cast (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    actor_id        INTEGER REFERENCES actor_metrics(actor_id),
    role_type       VARCHAR(50),   -- lead, supporting, cameo
    billing_order   INTEGER
);

-- 5. Regional theatrical window trends
CREATE TABLE IF NOT EXISTS theatrical_window_trends (
    id              SERIAL PRIMARY KEY,
    territory       VARCHAR(100),
    territory_code  CHAR(2),
    year            INTEGER,
    avg_theatrical_window_days INTEGER,  -- days before VOD/digital
    premium_vod_window_days    INTEGER,
    streaming_window_days      INTEGER,
    theatrical_revenue_share_pct NUMERIC(5,2),
    vod_revenue_share_pct       NUMERIC(5,2),
    streaming_revenue_share_pct NUMERIC(5,2),
    avg_screens_per_film        INTEGER,
    yoy_admissions_change_pct   NUMERIC(5,2),
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 6. Currency exchange rates
CREATE TABLE IF NOT EXISTS currency_exchange_rates (
    id              SERIAL PRIMARY KEY,
    currency_code   CHAR(3),
    currency_name   VARCHAR(100),
    territory       VARCHAR(100),
    usd_rate        NUMERIC(12,6),   -- 1 USD = X local currency
    eur_rate        NUMERIC(12,6),
    rate_date       DATE,
    volatility_index NUMERIC(5,2),  -- annualized FX volatility %
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 7. VOD/Streaming licensing price benchmarks
CREATE TABLE IF NOT EXISTS vod_licensing_benchmarks (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    platform        VARCHAR(100),   -- Netflix, Prime, Hulu, Disney+, etc.
    territory       VARCHAR(100),
    license_fee_usd BIGINT,
    deal_type       VARCHAR(50),    -- exclusive, non-exclusive, windowed
    license_term_months INTEGER,
    revenue_share_pct NUMERIC(5,2),
    streams_projected INTEGER,
    avg_revenue_per_stream_usd NUMERIC(6,4),
    deal_date       DATE,
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 8. Festival performance & awards
CREATE TABLE IF NOT EXISTS festival_performance (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    festival_name   VARCHAR(150),
    festival_year   INTEGER,
    award_category  VARCHAR(150),
    award_result    VARCHAR(50),    -- won, nominated, screened, special mention
    audience_score  NUMERIC(4,1),  -- out of 10
    critic_score    NUMERIC(4,1),
    buzz_score      INTEGER,       -- social mentions index 0-100
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 9. Genre-territory revenue benchmarks (historical averages)
CREATE TABLE IF NOT EXISTS genre_territory_benchmarks (
    id              SERIAL PRIMARY KEY,
    genre           VARCHAR(100),
    territory       VARCHAR(100),
    territory_code  CHAR(2),
    avg_opening_weekend_usd BIGINT,
    avg_total_gross_usd     BIGINT,
    avg_multiplier          NUMERIC(4,2),  -- total/opening ratio
    sample_size             INTEGER,
    year_range_start        INTEGER,
    year_range_end          INTEGER,
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 10. Marketing spend vs. revenue analysis
CREATE TABLE IF NOT EXISTS marketing_performance (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    territory       VARCHAR(100),
    p_and_a_spend_usd BIGINT,      -- prints & advertising
    digital_spend_usd BIGINT,
    tv_spend_usd    BIGINT,
    outdoor_spend_usd BIGINT,
    social_spend_usd  BIGINT,
    revenue_generated_usd BIGINT,
    roi_pct         NUMERIC(6,2),
    campaign_type   VARCHAR(100),
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 11. Censorship & content restriction flags
CREATE TABLE IF NOT EXISTS censorship_risk_flags (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    territory       VARCHAR(100),
    territory_code  CHAR(2),
    risk_level      VARCHAR(20),   -- low, medium, high, banned
    content_type    VARCHAR(100),  -- violence, nudity, political, religious, etc.
    description     TEXT,
    required_cuts   BOOLEAN DEFAULT FALSE,
    rating_assigned VARCHAR(20),
    approved        BOOLEAN,
    created_at      TIMESTAMP DEFAULT NOW()
);

-- 12. Acquisition deal history (comparable deals)
CREATE TABLE IF NOT EXISTS acquisition_deals (
    id              SERIAL PRIMARY KEY,
    film_id         INTEGER REFERENCES films(film_id),
    territory       VARCHAR(100),
    deal_type       VARCHAR(50),       -- theatrical, all-rights, split-rights
    minimum_guarantee_usd BIGINT,
    advance_usd     BIGINT,
    recoupment_threshold_usd BIGINT,
    backend_pct     NUMERIC(5,2),
    acquirer        VARCHAR(150),
    deal_date       DATE,
    outcome         VARCHAR(50),       -- profitable, breakeven, loss
    actual_revenue_usd BIGINT,
    created_at      TIMESTAMP DEFAULT NOW()
);

    f.year,
    f.genre,
    f.subgenre,
    f.director,
    f.production_budget_usd,
    f.mpaa_rating,
    f.studio,
    SUM(b.total_gross_usd)                            AS worldwide_gross_usd,
    SUM(b.total_gross_usd) / NULLIF(f.production_budget_usd,0) AS worldwide_multiple,
    COUNT(DISTINCT b.territory)                       AS territory_count,
    ROUND(AVG(fp.critic_score),2)                     AS avg_critic_score,
    ROUND(AVG(fp.audience_score),2)                   AS avg_audience_score,
    MAX(fp.buzz_score)                                AS peak_buzz_score,
    SUM(mp.p_and_a_spend_usd)                         AS total_marketing_spend_usd
FROM films f
LEFT JOIN box_office_performance b  ON f.film_id = b.film_id
LEFT JOIN festival_performance fp   ON f.film_id = fp.film_id
LEFT JOIN marketing_performance mp  ON f.film_id = mp.film_id
GROUP BY f.film_id, f.title, f.year, f.genre, f.subgenre, f.director,
         f.production_budget_usd, f.mpaa_rating, f.studio;

-- View: Territory risk + theatrical window combined
CREATE OR REPLACE VIEW vw_territory_acquisition_profile AS
SELECT
    r.territory,
    r.territory_code,
    r.overall_risk,
    r.market_attractiveness,
    r.censorship_risk,
    r.currency_risk,
    r.piracy_risk,
    r.collection_risk,
    r.notes                          AS risk_notes,
    w.avg_theatrical_window_days,
    w.theatrical_revenue_share_pct,
    w.vod_revenue_share_pct,
    w.streaming_revenue_share_pct,
    w.yoy_admissions_change_pct,
    x.currency_code,
    x.usd_rate,
    x.volatility_index
FROM territory_risk_index r
LEFT JOIN theatrical_window_trends w
       ON r.territory = w.territory AND w.year = (SELECT MAX(year) FROM theatrical_window_trends WHERE territory = r.territory)
LEFT JOIN currency_exchange_rates x
       ON r.territory = x.territory AND x.rate_date = (SELECT MAX(rate_date) FROM currency_exchange_rates WHERE territory = r.territory);

-- View: Actor power score per film
CREATE OR REPLACE VIEW vw_film_talent_power AS
SELECT
    f.film_id,
    f.title,
    COUNT(DISTINCT fc.actor_id)                                   AS cast_size,
    ROUND(AVG(a.q_score_global), 2)                               AS avg_q_score_global,
    MAX(a.q_score_global)                                         AS top_q_score_global,
    SUM(a.instagram_followers_m + a.twitter_followers_m + a.tiktok_followers_m) AS total_social_reach_m,
    SUM(a.franchise_value_usd)                                    AS total_franchise_value_usd,
    SUM(a.award_wins)                                             AS total_award_wins
FROM films f
JOIN film_cast fc  ON f.film_id = fc.film_id
JOIN actor_metrics a ON fc.actor_id = a.actor_id
GROUP BY f.film_id, f.title;
